Std modules

Std.Either

A two-branch sum type for computations that can return a left or right value.

Use

import Std.Either

Raw source: Either.hml

Datatype

datatype ('a, 'b) either = Left of 'a | Right of 'b

Public surface

  • map_right : ('b -> 'c) -> ('a, 'b) either -> ('a, 'c) either
  • map_left : ('a -> 'c) -> ('a, 'b) either -> ('c, 'b) either
  • is_left : ('a, 'b) either -> bool
  • is_right : ('a, 'b) either -> bool
  • from_left : 'a -> ('a, 'b) either -> 'a
  • from_right : 'b -> ('a, 'b) either -> 'b

Exhaustive example

import Std.Either

val left_value = Either.Left "boom"
val right_value = Either.Right 200

val left_mapped = Either.map_left (fn msg => msg ^ "!") left_value
val right_mapped = Either.map_right (fn code => code + 1) right_value

val left_check = Either.is_left left_mapped
val right_check = Either.is_right right_mapped

val left_result = Either.from_left "default" left_mapped
val right_result = Either.from_right 0 right_mapped

val summary =
  case right_mapped of
      Either.Left err => err
    | Either.Right code => int_to_string code
