Std modules

Std.List

A small list utility module for map/filter/fold style traversals and basic list inspection.

Use

import Std.List

Raw source: List.hml

Public surface

  • map : ('a -> 'b) -> 'a list -> 'b list
  • filter : ('a -> bool) -> 'a list -> 'a list
  • foldl : ('b * 'a -> 'b) -> 'b -> 'a list -> 'b
  • foldr : ('a * 'b -> 'b) -> 'b -> 'a list -> 'b
  • length : 'a list -> int
  • reverse : 'a list -> 'a list
  • append : 'a list -> 'a list -> 'a list
  • nth : 'a list -> int -> 'a
  • last : 'a list -> 'a
  • take : int -> 'a list -> 'a list
  • drop : int -> 'a list -> 'a list
  • zip : 'a list -> 'b list -> ('a * 'b) list
  • all : ('a -> bool) -> 'a list -> bool

Exhaustive example

import Std.List

val xs = [1, 2, 3, 4]
val doubled = List.map (fn x => x * 2) xs
val evens = List.filter (fn x => x mod 2 = 0) xs
val total = List.foldl (fn (acc, x) => acc + x) 0 xs
val rendered = List.foldr (fn (x, acc) => int_to_string x ^ acc) "" xs
val count = List.length xs
val reversed = List.reverse xs
val combined = List.append xs [5, 6]
val second = List.nth xs 1
val tail_last = List.last [2, 3, 4]
val first_two = List.take 2 xs
val rest = List.drop 2 xs
val pairs = List.zip ["a", "b"] [1, 2]
val all_even = List.all (fn x => x mod 2 = 0) evens
val any_even = List.any (fn x => x mod 2 = 0) xs
val found = List.find (fn x => x = 3) xs

val _ = println (int_to_string total)
