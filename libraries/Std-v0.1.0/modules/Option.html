Std modules

Std.Option

The standard optional value datatype and a few small helpers.

Use

import Std.Option

Raw source: Option.hml

Datatype

datatype 'a option = None | Some of 'a

Public surface

  • is_some : 'a option -> bool
  • is_none : 'a option -> bool
  • map_option : ('a -> 'b) -> 'a option -> 'b option
  • get_or : 'a option -> 'a -> 'a
  • flat_map_option : ('a -> 'b option) -> 'a option -> 'b option

Exhaustive example

import Std.Option

val some_value = Option.Some 41
val none_value = Option.None

val some_check = Option.is_some some_value
val none_check = Option.is_none none_value

val mapped = Option.map_option (fn x => x + 1) some_value
val fallback = Option.get_or none_value 0
val chained =
  Option.flat_map_option
    (fn x =>
      if x > 40 then Option.Some (x * 2)
      else Option.None)
    some_value

val result =
  case chained of
      Option.None => fallback
    | Option.Some x => x
