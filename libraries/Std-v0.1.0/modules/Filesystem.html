Std.Filesystem

Effect-based filesystem API built on the current raw filesystem builtins. Filesystem actions are performed inside Filesystem.run.

Use

import Std.Filesystem

Raw source: Filesystem.hml

Public surface

  • read_text : string -> string inside run
  • read_bytes : string -> bytes inside run
  • write_text : string -> string -> unit inside run
  • exists : string -> bool inside run
  • list_entries : string -> string list inside run
  • remove : string -> unit inside run
  • create_directory : string -> unit inside run
  • is_directory : string -> bool inside run
  • is_regular_file : string -> bool inside run
  • glob_matches : string -> string list inside run
  • walk : string -> string list inside run
  • join : string * string -> string pure
  • run : (unit -> 'a) -> 'a

Read/write example

import Std.Filesystem

val summary =
  Filesystem.run (fn () =>
    let
      val exists_before = Filesystem.exists "sample.txt"
      val text = Filesystem.read_text "sample.txt"
      val bytes = Filesystem.read_bytes "sample.txt"
      val digest = blake3 bytes
      val _ = Filesystem.write_text "copy.txt" text
      val exists_after = Filesystem.exists "copy.txt"
    in
      if exists_before andalso exists_after then digest else "missing"
    end)

val _ = println summary

Traversal and management example

import Std.Filesystem

val report =
  Filesystem.run (fn () =>
    let
      val _ = Filesystem.create_directory "tmp-out"
      val joined = Filesystem.join ("tmp-out", "nested.txt")
      val is_dir = Filesystem.is_directory "tmp-out"
      val is_file = Filesystem.is_regular_file joined
      val entries = Filesystem.list_entries "."
      val matched = Filesystem.glob_matches "*.txt"
      val walked = Filesystem.walk "."
      val _ =
        if Filesystem.exists joined then
          Filesystem.remove joined
        else
          ()
    in
      (joined, is_dir, is_file, entries, matched, walked)
    end)
